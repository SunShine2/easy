$.EasyTouch.Model，MVC中的Model模块，可以独立于EasyTouch运行

var Address = $.EasyTouch.Model.extend({
   server: {
       url: 'center/address/info/'
   },
   watch: {
       page: 'data.currentPage'
   },
   init: function(){

   },
   parse: function(json){
       return json.data;
   },
   validate: function(json){
       return json.code === 200;
   }

});
