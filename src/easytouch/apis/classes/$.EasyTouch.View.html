$.EasyTouch.View Class

Extends $.Base
Defined in: js\view.js:8
Module: EasyTouch-View

$.EasyTouch.View，MVC中的View模块，可以独立于EasyTouch运行

   var View = $.EasyTouch.View.extend({
       container: '<li></li>',
       events: {
           '.et-del': {
               tap: 'del'
           }
       },
       init: function(){
       },
       del: function(e){
       }
   });
   new View();
